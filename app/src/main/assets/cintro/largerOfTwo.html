/* Find larger value between two variable */ 

#include<stdio.h>
int main()
{
    int a,b;
    printf("Enter value of a: ");
    scanf("%d",&a);
    printf("Enter value of b: ");
    scanf("%d",&b);     
    if(a>b)
        printf("A is larger");
    else if(b>a)
        printf("B is larger");
    else
        printf("A and B are equal");
    return 0; 
}
